1. UI 단 설정
    * jquery UI framework 소스를 수정하는 문제가 이것 저것 생겨서 그냥 만들어봤습니다.
      일단은 내부 디자인에 맞춰서 대충 만들었으니 참고 하세요.

    * Tab 클릭시 이동은 href 속성에 정의된 링크를 이용한다.
      일반 url과 #id 를 이용할 수 있다.

      - url : ajax를 통한 html을 표시
      - #panelId : 미리 생성해둔 아래와 같은 페널을 활성화 한다.
        예 : <div id="panelId" style="display:none;"></div>
            
    - UI 라이브러리 로딩
        <op:jsTag type="openworks" items="ui" />

    -   $().ready() 함수 내에 아래 셈플을 수정하여 적용

        $().ready(function() {
            /** 옵션 설정 : 미설정시 기본값 사용 */
            var options = {
                    // 내용을 표시할 패널을 지정. 생략 가능
                    tabContainer : "tabContainer",
                    // 공통 파라미터를 지정할 수 있음. 생략 가능
                    baseParam: {
                        domainCd: "${dataVo.domainCd}",
                        menuCode: "${dataVo.menuCode}"
                    },
                    // 로드전 이벤트 정의. 생략 가능
                    beforeLoad : function(event, item) {

                        return true;
                    },
                    // 로드 후 이벤트 정의. 생략 가능
                    afterLoad : function(event, item) {

                        return true;
                    }
            };
            /** 대상 input tag에 Tabs 설정*/
            $tabs = $("#selector").jsTabUi(options);
        }

    - 사용가능 메소드
     
        /** 현재 Tab 새로고침 */
        $tabs.refresh();
        
        /** 지정한 index 순서에 따른 Tab 활성화 */
        tabs.moveTab(index);
        
        /** 지정한 index 순서에 해당하는 텝을 새로 표시 */
        tabs.loadTab(index);

        /** 현재 선택되어 있는 tab의 일련번호를 반환 */
        var index = tabs.currentIndex();

        /** 지정한 일련번호에 해당하는 tab의 url을 변경 */
        tabs.changeUrl(index, url);

        /** 전체 tab을 비선택 상태로 초기화 */
        tabs.clearTab();
